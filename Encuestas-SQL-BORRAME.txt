-- 1. Tabla de Encuestas
CREATE TABLE IF NOT EXISTS public.encuestas (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  pregunta text NOT NULL,
  opciones text[] NOT NULL, -- Array de textos para las opciones
  activa boolean DEFAULT true,
  created_at timestamp with time zone DEFAULT now(),
  creador_id uuid REFERENCES auth.users(id)
);

-- 2. Tabla de Votos
CREATE TABLE IF NOT EXISTS public.votos_encuesta (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  encuesta_id uuid REFERENCES public.encuestas(id) ON DELETE CASCADE,
  usuario_id uuid REFERENCES auth.users(id) ON DELETE CASCADE,
  opcion_index integer NOT NULL, -- El índice de la opción elegida
  created_at timestamp with time zone DEFAULT now(),
  UNIQUE(encuesta_id, usuario_id) -- Impide que un usuario vote dos veces la misma encuesta
);

-- 3. Habilitar RLS
ALTER TABLE public.encuestas ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.votos_encuesta ENABLE ROW LEVEL SECURITY;

-- 4. Políticas para Encuestas
CREATE POLICY "Todo el mundo puede ver encuestas activas" 
ON public.encuestas FOR SELECT USING (true);

CREATE POLICY "Solo admins pueden gestionar encuestas" 
ON public.encuestas FOR ALL TO authenticated 
USING (EXISTS (SELECT 1 FROM public.profiles WHERE id = auth.uid() AND is_admin = true));

-- 5. Políticas para Votos
CREATE POLICY "Usuarios pueden ver todos los votos" 
ON public.votos_encuesta FOR SELECT USING (true);

CREATE POLICY "Usuarios pueden votar" 
ON public.votos_encuesta FOR INSERT TO authenticated 
WITH CHECK (auth.uid() = usuario_id);
