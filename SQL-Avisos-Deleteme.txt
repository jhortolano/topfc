create table avisos (
  id bigint primary key default 1, -- Usamos 1 para manejar un solo aviso global
  titulo text,
  contenido text,
  mostrar boolean default false,
  updated_at timestamp with time zone default now()
);

-- Insertamos la fila inicial si no existe
insert into avisos (id, titulo, contenido, mostrar) 
values (1, '', '', false)
on conflict (id) do nothing;


-- 1. Habilitamos la seguridad
ALTER TABLE public.avisos ENABLE ROW LEVEL SECURITY;

-- 2. Permitimos que CUALQUIER usuario logueado pueda gestionar la tabla
CREATE POLICY "Acceso total para usuarios autenticados"
ON public.avisos
FOR ALL 
TO authenticated
USING (true)
WITH CHECK (true);

-- 3. Permitimos que todo el mundo (incluso sin loguear) pueda LEER si mostrar es true
CREATE POLICY "Lectura pública si está activo"
ON public.avisos
FOR SELECT
TO anon, authenticated
USING (mostrar = true);
