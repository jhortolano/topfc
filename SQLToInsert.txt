-- ============================================================
-- ACTUALIZACIÓN DE VISTAS: soporte para PNJ (Partidos No Jugados)
--
-- Un partido es "no jugado" para un jugador cuando:
--   - matches.is_played = FALSE
--   - su propio score en ese partido es 0
--     (home_score si es home_team, away_score si es away_team)
--
-- Efecto sobre la clasificación:
--   - PNJ  += 1  (en lugar de PJ)
--   - PTS  += -1 (penalización, incluso si el resultado sería empate 0-0)
-- ============================================================


-- ============================================================
-- 1. VISTA: partidos_detallados
--    Se añade la columna "es_no_jugado_local" y "es_no_jugado_visitante"
--    para que sea fácil detectarlo desde el front si se necesita.
-- ============================================================

DROP VIEW IF EXISTS public.partidos_detallados CASCADE;

CREATE VIEW public.partidos_detallados
WITH (security_invoker = 'on') AS
SELECT
    m.id,
    m.home_team,
    m.away_team,
    m.home_score,
    m.away_score,
    m.is_played,
    m.week,
    m.season,
    m.division,
    m.created_at,
    ms.stream_url,
    ph.nick          AS local_nick,
    ph.avatar_url    AS local_avatar,
    pa.nick          AS visitante_nick,
    pa.avatar_url    AS visitante_avatar,
    m.issues,
    -- Partido no jugado para el jugador LOCAL
    (m.is_played = FALSE AND m.home_score = 0)  AS es_no_jugado_local,
    -- Partido no jugado para el jugador VISITANTE
    (m.is_played = FALSE AND m.away_score = 0)  AS es_no_jugado_visitante
FROM public.matches m
LEFT JOIN public.profiles  ph ON m.home_team = ph.id
LEFT JOIN public.profiles  pa ON m.away_team = pa.id
LEFT JOIN public.match_streams ms ON m.id    = ms.match_id;


-- ============================================================
-- 2. VISTA: clasificacion
--    Se añade la columna "pnj" (partidos no jugados).
--    - pj  solo cuenta los partidos donde el jugador SÍ jugó.
--    - pnj cuenta los partidos donde el jugador NO jugó.
--    - pts aplica -1 por cada partido no jugado (en lugar de
--      sumar los puntos del resultado).
-- ============================================================

DROP VIEW IF EXISTS public.clasificacion CASCADE;

CREATE VIEW public.clasificacion
WITH (security_invoker = 'true') AS
WITH lista_jugadores AS (
    -- Todos los jugadores que aparecen en algún partido de cada temporada/división
    SELECT DISTINCT home_team AS user_id, season, division FROM public.matches
    UNION
    SELECT DISTINCT away_team AS user_id, season, division FROM public.matches
),
stats_partidos AS (
    SELECT
        m.season,
        m.division,
        m.home_team,
        m.away_team,
        m.home_score,
        m.away_score,
        m.is_played,

        -- ¿Es partido no jugado para el LOCAL?
        -- Condición: is_played = FALSE AND home_score = 0
        (m.is_played = FALSE AND m.home_score = 0 AND m.away_score IS NOT NULL)  AS no_jugado_h,

        -- ¿Es partido no jugado para el VISITANTE?
        -- Condición: is_played = FALSE AND away_score = 0
        (m.is_played = FALSE AND m.away_score = 0 AND m.home_score IS NOT NULL)  AS no_jugado_a,

        -- Puntos normales (solo si el partido SÍ fue jugado por ese lado)
        -- Para LOCAL
        CASE
            WHEN (m.is_played = FALSE AND m.home_score = 0 AND m.away_score IS NOT NULL)
                THEN -1   -- penalización por no jugar
            WHEN m.home_score > m.away_score  THEN 3
            WHEN m.home_score = m.away_score  THEN 1
            ELSE 0
        END AS pts_h,

        -- Para VISITANTE
        CASE
            WHEN (m.is_played = FALSE AND m.away_score = 0 AND m.home_score IS NOT NULL)
                THEN -1   -- penalización por no jugar
            WHEN m.away_score > m.home_score  THEN 3
            WHEN m.home_score = m.away_score  THEN 1
            ELSE 0
        END AS pts_a,

        -- Victorias (solo si jugó)
        CASE
            WHEN (m.is_played = FALSE AND m.home_score = 0) THEN 0
            WHEN m.home_score > m.away_score THEN 1
            ELSE 0
        END AS win_h,

        CASE
            WHEN (m.is_played = FALSE AND m.away_score = 0) THEN 0
            WHEN m.away_score > m.home_score THEN 1
            ELSE 0
        END AS win_a,

        -- Empates (solo si jugó; un 0-0 con is_played=FALSE es no jugado, no empate)
        CASE
            WHEN (m.is_played = FALSE AND (m.home_score = 0 OR m.away_score = 0)) THEN 0
            WHEN m.home_score = m.away_score THEN 1
            ELSE 0
        END AS draw

    FROM public.matches m
    WHERE m.home_score IS NOT NULL
      AND m.away_score IS NOT NULL
)
SELECT
    lj.user_id,
    lj.season,
    lj.division,
    p.nick,
    p.avatar_url,

    -- PTS: suma de puntos (incluyendo -1 por PNJ)
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN sp.pts_h
            WHEN sp.away_team = lj.user_id THEN sp.pts_a
            ELSE 0
        END
    ), 0) AS pts,

    -- PJ: partidos efectivamente jugados (excluye los PNJ)
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id AND NOT sp.no_jugado_h THEN 1
            WHEN sp.away_team = lj.user_id AND NOT sp.no_jugado_a THEN 1
            ELSE 0
        END
    ), 0) AS pj,

    -- PNJ: partidos no jugados
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id AND sp.no_jugado_h THEN 1
            WHEN sp.away_team = lj.user_id AND sp.no_jugado_a THEN 1
            ELSE 0
        END
    ), 0) AS pnj,

    -- PG: victorias
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN sp.win_h
            WHEN sp.away_team = lj.user_id THEN sp.win_a
            ELSE 0
        END
    ), 0) AS pg,

    -- PE: empates
    COALESCE(SUM(
        CASE
            WHEN (sp.home_team = lj.user_id OR sp.away_team = lj.user_id) THEN sp.draw
            ELSE 0
        END
    ), 0) AS pe,

    -- GF: goles a favor
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN sp.home_score
            WHEN sp.away_team = lj.user_id THEN sp.away_score
            ELSE 0
        END
    ), 0) AS gf,

    -- GC: goles en contra
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN sp.away_score
            WHEN sp.away_team = lj.user_id THEN sp.home_score
            ELSE 0
        END
    ), 0) AS gc,

    -- DG: diferencia de goles
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN (sp.home_score - sp.away_score)
            WHEN sp.away_team = lj.user_id THEN (sp.away_score - sp.home_score)
            ELSE 0
        END
    ), 0) AS dg

FROM lista_jugadores lj
LEFT JOIN public.profiles p        ON lj.user_id = p.id
LEFT JOIN stats_partidos sp
       ON (lj.user_id = sp.home_team OR lj.user_id = sp.away_team)
      AND lj.season   = sp.season
      AND lj.division = sp.division
GROUP BY
    lj.user_id, lj.season, lj.division, p.nick, p.avatar_url
ORDER BY
    -- Orden: PTS desc, DG desc, GF desc
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN sp.pts_h
            WHEN sp.away_team = lj.user_id THEN sp.pts_a
            ELSE 0
        END
    ), 0) DESC,
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN (sp.home_score - sp.away_score)
            WHEN sp.away_team = lj.user_id THEN (sp.away_score - sp.home_score)
            ELSE 0
        END
    ), 0) DESC,
    COALESCE(SUM(
        CASE
            WHEN sp.home_team = lj.user_id THEN sp.home_score
            WHEN sp.away_team = lj.user_id THEN sp.away_score
            ELSE 0
        END
    ), 0) DESC;

-- Restaurar permisos (igual que el dump original)
GRANT ALL ON TABLE public.clasificacion      TO anon;
GRANT ALL ON TABLE public.clasificacion      TO authenticated;
GRANT ALL ON TABLE public.clasificacion      TO service_role;

GRANT ALL ON TABLE public.partidos_detallados TO anon;
GRANT ALL ON TABLE public.partidos_detallados TO authenticated;
GRANT ALL ON TABLE public.partidos_detallados TO service_role;













-- ============================================================
-- ACTUALIZACIÓN DE VISTA: extra_playoffs_clasificacion
-- Soporte para PNJ (Partidos No Jugados)
--
-- Un partido es "no jugado" para un jugador cuando:
--   - extra_matches.stream_url = 'https://www.twitch.tv/p/es-es/about/'
--   - su propio score es 0:
--       score1 si es player1_id
--       score2 si es player2_id
--
-- Efecto:
--   - PNJ += 1  (en lugar de PJ)
--   - PTS += -1 (penalización)
-- ============================================================

-- Usamos DROP + CREATE para evitar el error de reordenación de columnas
DROP VIEW IF EXISTS public.extra_playoffs_clasificacion CASCADE;

CREATE VIEW public.extra_playoffs_clasificacion
WITH (security_invoker = 'true') AS
WITH partidos_normalizados AS (

    -- Perspectiva de PLAYER 1
    SELECT
        em.extra_id,
        em.group_id,
        em.player1_id                                                        AS p_id,
        em.score1                                                            AS gf,
        em.score2                                                            AS gc,

        -- ¿Es partido no jugado para player1?
        (em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score1 = 0)  AS es_pnj,

        -- PG
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score1 = 0 THEN 0
            WHEN em.score1 > em.score2 THEN 1
            ELSE 0
        END AS pg,

        -- PE
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score1 = 0 THEN 0
            WHEN em.score1 = em.score2 THEN 1
            ELSE 0
        END AS pe,

        -- PP
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score1 = 0 THEN 0
            WHEN em.score1 < em.score2 THEN 1
            ELSE 0
        END AS pp,

        -- Puntos: -1 si PNJ, normal si no
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score1 = 0 THEN -1
            WHEN em.score1 > em.score2 THEN 3
            WHEN em.score1 = em.score2 THEN 1
            ELSE 0
        END AS pts_calculados

    FROM public.extra_matches em
    WHERE em.group_id IS NOT NULL
      AND em.score1 IS NOT NULL
      AND em.score2 IS NOT NULL

    UNION ALL

    -- Perspectiva de PLAYER 2
    SELECT
        em.extra_id,
        em.group_id,
        em.player2_id                                                        AS p_id,
        em.score2                                                            AS gf,
        em.score1                                                            AS gc,

        -- ¿Es partido no jugado para player2?
        (em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score2 = 0)  AS es_pnj,

        -- PG
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score2 = 0 THEN 0
            WHEN em.score2 > em.score1 THEN 1
            ELSE 0
        END AS pg,

        -- PE
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score2 = 0 THEN 0
            WHEN em.score2 = em.score1 THEN 1
            ELSE 0
        END AS pe,

        -- PP
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score2 = 0 THEN 0
            WHEN em.score2 < em.score1 THEN 1
            ELSE 0
        END AS pp,

        -- Puntos: -1 si PNJ, normal si no
        CASE
            WHEN em.stream_url = 'https://www.twitch.tv/p/es-es/about/' AND em.score2 = 0 THEN -1
            WHEN em.score2 > em.score1 THEN 3
            WHEN em.score2 = em.score1 THEN 1
            ELSE 0
        END AS pts_calculados

    FROM public.extra_matches em
    WHERE em.group_id IS NOT NULL
      AND em.score1 IS NOT NULL
      AND em.score2 IS NOT NULL

),
universo_jugadores AS (
    SELECT DISTINCT extra_id, group_id, player1_id AS user_id
    FROM public.extra_matches
    WHERE group_id IS NOT NULL
    UNION
    SELECT DISTINCT extra_id, group_id, player2_id AS user_id
    FROM public.extra_matches
    WHERE group_id IS NOT NULL
)
SELECT
    uj.extra_id                                                          AS playoff_extra_id,
    uj.group_id,
    COALESCE(g.nombre_grupo, 'Grupo ' || uj.group_id::text)             AS nombre_grupo_texto,
    p.nick,
    p.avatar_url,
    p.id                                                                 AS user_id,

    -- PJ: solo partidos efectivamente jugados
    COALESCE(SUM(CASE WHEN pn.p_id IS NOT NULL AND NOT pn.es_pnj THEN 1 ELSE 0 END), 0)  AS pj,

    -- PNJ: partidos no jugados
    COALESCE(SUM(CASE WHEN pn.es_pnj THEN 1 ELSE 0 END), 0)            AS pnj,

    -- PG, PE, PP
    COALESCE(SUM(pn.pg), 0)                                             AS pg,
    COALESCE(SUM(pn.pe), 0)                                             AS pe,
    COALESCE(SUM(pn.pp), 0)                                             AS pp,

    -- PTS (incluye -1 por PNJ)
    COALESCE(SUM(pn.pts_calculados), 0)                                 AS pts,

    -- GF, GC, DG
    COALESCE(SUM(pn.gf), 0)                                             AS gf,
    COALESCE(SUM(pn.gc), 0)                                             AS gc,
    COALESCE(SUM(pn.gf) - SUM(pn.gc), 0)                               AS dg

FROM universo_jugadores uj
JOIN  public.profiles   p  ON uj.user_id  = p.id
JOIN  public.extra_groups g ON uj.group_id = g.id
LEFT JOIN partidos_normalizados pn
       ON uj.extra_id = pn.extra_id
      AND uj.user_id  = pn.p_id
GROUP BY
    uj.extra_id, uj.group_id, g.nombre_grupo, p.id, p.nick, p.avatar_url;

-- Restaurar permisos
GRANT ALL ON TABLE public.extra_playoffs_clasificacion TO anon;
GRANT ALL ON TABLE public.extra_playoffs_clasificacion TO authenticated;
GRANT ALL ON TABLE public.extra_playoffs_clasificacion TO service_role;
