-- 1. Tabla para los periodos/fechas de las promociones
CREATE TABLE IF NOT EXISTS weeks_promo (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    season INT NOT NULL,
    start_at TIMESTAMPTZ NOT NULL,
    end_at TIMESTAMPTZ NOT NULL,
    idavuelta TEXT CHECK (idavuelta IN ('ida', 'vuelta')),
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. Tabla para los enfrentamientos individuales
CREATE TABLE IF NOT EXISTS promo_matches (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    player1_id UUID REFERENCES profiles(id),
    player2_id UUID REFERENCES profiles(id),
    score1 INT DEFAULT 0,
    score2 INT DEFAULT 0,
    is_played BOOLEAN DEFAULT FALSE,
    season INT NOT NULL,
    division INT NOT NULL,
    idavuelta TEXT CHECK (idavuelta IN ('ida', 'vuelta')),
    label_info TEXT, -- Almacena info de posiciones (ej: "Pos 8 Div 1 vs Pos 3 Div 2")
    stream_url TEXT, -- URL de la retransmisión del partido
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);



-- Activar RLS en las tablas
ALTER TABLE weeks_promo ENABLE ROW LEVEL SECURITY;
ALTER TABLE promo_matches ENABLE ROW LEVEL SECURITY;

-- Política: Todos pueden ver las fechas de promoción y los partidos
CREATE POLICY "Permitir lectura pública de promociones" 
ON weeks_promo FOR SELECT USING (true);

CREATE POLICY "Permitir lectura pública de partidos de promo" 
ON promo_matches FOR SELECT USING (true);

-- Política: Solo administradores pueden insertar/actualizar (basado en tu tabla profiles)
CREATE POLICY "Admin full access weeks_promo" 
ON weeks_promo FOR ALL 
TO authenticated 
USING ((SELECT is_admin FROM profiles WHERE id = auth.uid()) = true);

CREATE POLICY "Admin full access promo_matches" 
ON promo_matches FOR ALL 
TO authenticated 
USING ((SELECT is_admin FROM profiles WHERE id = auth.uid()) = true);
