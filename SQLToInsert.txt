CREATE TABLE IF NOT EXISTS diccionario_equipos (
    -- Identificador único autoincremental
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    
    -- Booleans con valor por defecto False
    es_liga BOOLEAN DEFAULT FALSE NOT NULL,
    es_playoff BOOLEAN DEFAULT FALSE NOT NULL,
    es_extra_playoff BOOLEAN DEFAULT FALSE NOT NULL,
    
    -- Enteros
    season INTEGER,
    id_playoff INTEGER,
    
    -- Identificador de usuario (UUID suele ser el estándar en Supabase Auth)
    user_id UUID,
    
    -- Strings (Text permite longitud variable sin límite estricto)
    nick TEXT,
    texto1 TEXT,
    texto2 TEXT,
    texto3 TEXT,
    
    -- Timestamps (Opcional, pero recomendado en Supabase)
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Habilitar Row Level Security (RLS) es una buena práctica en Supabase
ALTER TABLE diccionario_equipos ENABLE ROW LEVEL SECURITY;


-- Convertimos la columna id_playoff de INTEGER a UUID
ALTER TABLE diccionario_equipos 
ALTER COLUMN id_playoff TYPE UUID USING id_playoff::text::uuid;



-- 1. Permitir que cualquier usuario lea todos los registros (Lectura pública)
CREATE POLICY "Permitir lectura pública" 
ON diccionario_equipos 
FOR SELECT 
USING (true);

-- 2. Permitir que los usuarios autenticados inserten sus propios datos
CREATE POLICY "Permitir inserción a usuarios autenticados" 
ON diccionario_equipos 
FOR INSERT 
TO authenticated 
WITH CHECK (auth.uid() = user_id);

-- 3. Permitir que los usuarios actualicen solo sus propios registros
CREATE POLICY "Permitir actualización a dueños del registro" 
ON diccionario_equipos 
FOR UPDATE 
TO authenticated 
USING (auth.uid() = user_id)
WITH CHECK (auth.uid() = user_id);

-- 4. Permitir que los usuarios borren solo sus propios registros
CREATE POLICY "Permitir borrado a dueños del registro" 
ON diccionario_equipos 
FOR DELETE 
TO authenticated 
USING (auth.uid() = user_id);


ALTER TABLE playoffs_extra 
ADD COLUMN tiene_diccionario BOOLEAN DEFAULT FALSE;


-- Eliminar políticas antiguas para evitar conflictos
DROP POLICY IF EXISTS "Permitir inserción a usuarios autenticados" ON diccionario_equipos;
DROP POLICY IF EXISTS "Permitir actualización a dueños del registro" ON diccionario_equipos;
DROP POLICY IF EXISTS "Permitir borrado a dueños del registro" ON diccionario_equipos;

-- Nueva política: Cualquier usuario autenticado (Admin) puede gestionar los registros
-- (O puedes restringirlo más si tienes un rol de admin específico)
CREATE POLICY "Gestión total para autenticados" 
ON diccionario_equipos 
FOR ALL 
TO authenticated 
USING (true) 
WITH CHECK (true);
